Array.prototype.square=function(){ return this.map(function(num) { return num * num; }); } Array.prototype.cube=function(){ return this.map(function(num) { return num * num*num; }); } Array.prototype.average=function(){ var s = 0; for (i = 0; i < this.length; i++){ s += this[i]; } return s/this.length; } Array.prototype.sum=function(){ var s = 0; for (i = 0; i < this.length; i++){ s += this[i]; } return s; } Array.prototype.even=function(){ let arr = []; for (i=0; i < this.length; i++){ if(this[i] % 2 == 0) { arr.push(this[i]); } } return arr; } Array.prototype.odd=function(){ let arr = []; for (i=0; i < this.length; i++){ if(this[i] % 2 !== 0) { arr.push(this[i]); } } return arr; }
